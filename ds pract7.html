print("hashing technique")
m=int(input("size of the hash table:"))
n=int(input("number of keys:"))
h=[]
for i in range(m):
h.append(0)
print("enter elements:)
for i in range(n):
key=int(input())
rel_add=key%m
if h[rel_add]==0:
h[rel_add]=key
print("relative address=",rel_add)
else:
print("Relative address=",rel_add,"(collision case)")
print("\nhash table")
print("index\tKey")
for i in range(m):
print(i,"\t",h[i])
