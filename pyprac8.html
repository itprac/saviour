import math

def circleArea(r):
return math.pi * r * r

def squareArea(a):
return a ** 2

def pointyShapeVolume(x, y, squareBase):
if squareBase:
base = squareArea(x) * (y / 3)
return base
else:
base = circleArea(x) * y / 3
return base

print(pointyShapeVolume(2, 3, True))
