reversing an array

print("reversing an array")
n=int(input("enter the number of elements:"))
A=[]
R_A=[]
print("enter the elements:")
for i in range(n):
elt=int(input(""))
A.append(elt)
R_A.append(0)
print("array:",A)
for i in range(n):
R_A[i]=A[n-(i+1)]
print("reversed array:",R_A)
