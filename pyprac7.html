class Shape:
author = 'Ashwin Mehta'

def __init__(self, x, y):
self.x = x
self.y = y

def area(self):
a = self.x * self.y
print('Area of a rectangle:', a)
print(self.author)

class Square(Shape):
def __init__(self, x):
super().__init__(x, x)

def area(self):
a = self.x * self.x
print('Area of a square:', a)
print(self.author)

r = Shape(12, 34)
r.area()

s = Square(34)
s.area()
