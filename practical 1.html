1.Bubble sort

print("Bubble Sort")
A = [7, 6, 4, 3, 0]
n=len (A)
print("A=",A)

for p in range(n-1):
print("Pass-",p+1)
for i in range (n-p-1):
if A[i] > A[i + 1]:
temp=A[i]
A[i]=A[i+1]
A[i+1]=temp
print("\tA=",A)

3.insertion sort
print("\nInsertion Sort")
A=[5,-1,2,7,3]
n=len(A)
print("A=",A)
for i in range(1,n):
temp=A[i]
k=i-1
while temp-1:
A[k+1]=A[k]
k=k-1
A[k+1]=temp
print("Pass-",i)
print("\tA=",A)
